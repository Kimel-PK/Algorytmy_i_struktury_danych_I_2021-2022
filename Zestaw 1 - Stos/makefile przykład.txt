CC        := gcc
CFLAGS    := -Wall -std=c99 -pedantic -pthread
LDFLAGS    := -pthread -L.
LDLIBS    := -lrt -lpthread

.PHONY: all clean
all: wykluczanie.x

%.x: %.o
    $(CC) $(LDFLAGS) -o $@ $^ $(LDLIBS)
%.o: %.c
    $(CC) $(CFLAGS) -c -o $@ $<

wykluczanie.x: wykluczanie.o
wykluczanie.o: wykluczanie.c


clean:
    $(RM) *.[aox]
    $(RM) *.txt
